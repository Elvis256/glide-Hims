On-Premise Deployment

Deploy Glide HIMS on your own servers

✓ Verified in v1.0 — public self-registration is off by default on production on-prem deployments. New tenants are provisioned by a system administrator from the System Administration portal; self-registration can be enabled by your administrator when appropriate. Tenant users sign in at their tenant login page — your administrator will share the URL during onboarding.
Prerequisites
Hardware Requirements
CPU4+ cores (8 recommended)
RAM8GB minimum (16GB recommended)
Storage100GB+ SSD
NetworkStatic IP, open ports 80/443
Software Requirements
OSUbuntu 22.04 LTS (recommended)
Node.js18.x or 20.x LTS
PostgreSQL15.x
Redis7.x (optional)
1
Prepare the Server
# Update system
sudo apt update && sudo apt upgrade -y

# Install essential packages
sudo apt install -y curl git build-essential nginx certbot python3-certbot-nginx

# Install Node.js 20.x
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt install -y nodejs

# Install pnpm
npm install -g pnpm

# Install PM2 for process management
npm install -g pm2

# Install PostgreSQL
sudo apt install -y postgresql postgresql-contrib

# Install Redis (optional but recommended)
sudo apt install -y redis-server
sudo systemctl enable redis-server
2
Configure Database
# Switch to postgres user
sudo -u postgres psql

# Create database and user
CREATE USER glide_admin WITH PASSWORD 'your_secure_password';
CREATE DATABASE glide_hims OWNER glide_admin;
GRANT ALL PRIVILEGES ON DATABASE glide_hims TO glide_admin;
\q

# Enable remote connections (if needed)
sudo nano /etc/postgresql/15/main/postgresql.conf
# Set: listen_addresses = '*'

sudo nano /etc/postgresql/15/main/pg_hba.conf
# Add: host glide_hims glide_admin 0.0.0.0/0 md5

# Restart PostgreSQL
sudo systemctl restart postgresql
3
Install Application
# Create application directory
sudo mkdir -p /opt/glide-hims
sudo chown $USER:$USER /opt/glide-hims
cd /opt/glide-hims

# Clone repository (or extract release package)
git clone https://github.com/itsolutionsuganda/glide-hims.git .
git checkout v1.0.0

# Install dependencies
pnpm install

# Build applications
pnpm run build
4
Configure Environment
# Create environment file
cat > .env << 'EOF'
# Server Configuration
NODE_ENV=production
PORT=3000

# Database
DB_HOST=localhost
DB_PORT=5432
DB_NAME=glide_hims
DB_USER=glide_admin
DB_PASSWORD=your_secure_password

# Security
JWT_SECRET=generate-a-256-bit-random-string-here
JWT_REFRESH_SECRET=another-256-bit-random-string-here
JWT_EXPIRATION=3600
JWT_REFRESH_EXPIRATION=604800

# Enterprise License
LICENSE_KEY=GLIDE-XXXX-XXXX-XXXX-XXXX
DEPLOYMENT_MODE=on-premise

# Phone Home (for license validation)
PHONE_HOME_URL=https://hmisdemo.itsolutionsuganda.com
PHONE_HOME_INTERVAL=86400

# Redis (optional)
REDIS_HOST=localhost
REDIS_PORT=6379

# File Storage
UPLOAD_DIR=/opt/glide-hims/uploads
MAX_FILE_SIZE=52428800

# CORS (adjust for your domain)
CORS_ORIGIN=https://hims.yourcompany.com
EOF

# Generate secure JWT secrets
openssl rand -base64 32  # Use output for JWT_SECRET
openssl rand -base64 32  # Use output for JWT_REFRESH_SECRET
5
Run Database Migrations
# Navigate to backend
cd packages/backend

# Run migrations
pnpm run migration:run

# Seed initial data (optional)
pnpm run seed
6
Configure PM2 Process Manager
# Create PM2 ecosystem file
cat > ecosystem.config.js << 'EOF'
module.exports = {
  apps: [
    {
      name: 'glide-hims-backend',
      cwd: '/opt/glide-hims/packages/backend',
      script: 'dist/main.js',
      instances: 'max',
      exec_mode: 'cluster',
      env: {
        NODE_ENV: 'production',
      },
      error_file: '/var/log/glide-hims/backend-error.log',
      out_file: '/var/log/glide-hims/backend-out.log',
      time: true,
    },
    {
      name: 'glide-hims-frontend',
      cwd: '/opt/glide-hims/packages/frontend',
      script: 'serve',
      args: '-s dist -l 5173',
      env: {
        NODE_ENV: 'production',
      },
      error_file: '/var/log/glide-hims/frontend-error.log',
      out_file: '/var/log/glide-hims/frontend-out.log',
    }
  ]
};
EOF

# Create log directory
sudo mkdir -p /var/log/glide-hims
sudo chown $USER:$USER /var/log/glide-hims

# Start applications
pm2 start ecosystem.config.js

# Save PM2 configuration
pm2 save

# Setup PM2 to start on boot
pm2 startup
# Run the command it outputs
7
Configure Nginx Reverse Proxy
# Create Nginx configuration
sudo cat > /etc/nginx/sites-available/glide-hims << 'EOF'
server {
    listen 80;
    server_name hims.yourcompany.com;
    
    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name hims.yourcompany.com;

    # SSL certificates (will be configured by certbot)
    ssl_certificate /etc/letsencrypt/live/hims.yourcompany.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/hims.yourcompany.com/privkey.pem;
    
    # SSL settings
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    # API requests
    location /api {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 86400;
    }

    # Frontend
    location / {
        proxy_pass http://localhost:5173;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # Static file caching
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
        proxy_pass http://localhost:5173;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}
EOF

# Enable site
sudo ln -s /etc/nginx/sites-available/glide-hims /etc/nginx/sites-enabled/

# Test configuration
sudo nginx -t

# Reload Nginx
sudo systemctl reload nginx
8
Install SSL Certificate
# Get SSL certificate from Let's Encrypt
sudo certbot --nginx -d hims.yourcompany.com

# Test auto-renewal
sudo certbot renew --dry-run
Tip: For internal networks without internet access, use a self-signed certificate or your organization's CA.
9
Configure Firewall
# Allow necessary ports
sudo ufw allow ssh
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp

# Enable firewall
sudo ufw enable

# Check status
sudo ufw status
10
Verify Installation
# Check services
pm2 status

# Test API
curl https://hims.yourcompany.com/api/v1/health

# Test updates endpoint
curl https://hims.yourcompany.com/api/v1/updates/check?version=0.0.0

# Check license
curl -X POST https://hims.yourcompany.com/api/v1/license/validate \
  -H "Content-Type: application/json" \
  -d '{"licenseKey": "YOUR-LICENSE-KEY"}'
Phone Home Configuration

On-premise installations periodically contact the license server to validate the license and check for updates. This requires outbound internet access to:

  • https://hmisdemo.itsolutionsuganda.com (license validation)
If internet access is restricted:
# Allow specific hosts through firewall
sudo ufw allow out to any port 443

# Or configure proxy
export HTTPS_PROXY=http://proxy.company.com:8080

# Add to .env
HTTPS_PROXY=http://proxy.company.com:8080
Important: If phone-home fails for 30 consecutive days, some enterprise features may be disabled. Contact support for offline license options.
Backup Configuration
# Create backup script
cat > /opt/glide-hims/backup.sh << 'EOF'
#!/bin/bash
BACKUP_DIR="/var/backups/glide-hims"
DATE=$(date +%Y%m%d_%H%M%S)
mkdir -p $BACKUP_DIR

# Database backup
pg_dump -U glide_admin -h localhost glide_hims | gzip > $BACKUP_DIR/db_$DATE.sql.gz

# Application files backup
tar -czf $BACKUP_DIR/uploads_$DATE.tar.gz /opt/glide-hims/uploads

# Keep only last 7 days
find $BACKUP_DIR -type f -mtime +7 -delete

echo "Backup completed: $DATE"
EOF

chmod +x /opt/glide-hims/backup.sh

# Add to crontab (daily at 2 AM)
(crontab -l 2>/dev/null; echo "0 2 * * * /opt/glide-hims/backup.sh") | crontab -
Systemd Service Files (Alternative to PM2)
# Backend service
sudo cat > /etc/systemd/system/glide-hims-backend.service << 'EOF'
[Unit]
Description=Glide HIMS Backend
After=network.target postgresql.service

[Service]
Type=simple
User=glide
WorkingDirectory=/opt/glide-hims/packages/backend
ExecStart=/usr/bin/node dist/main.js
Restart=on-failure
RestartSec=10
EnvironmentFile=/opt/glide-hims/.env

[Install]
WantedBy=multi-user.target
EOF

# Enable and start
sudo systemctl daemon-reload
sudo systemctl enable glide-hims-backend
sudo systemctl start glide-hims-backend
