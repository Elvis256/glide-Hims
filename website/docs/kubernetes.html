Kubernetes Deployment

Production-grade orchestration with high availability and auto-scaling

Best for: Large deployments, high availability requirements, multi-region deployments, and organizations already using Kubernetes.

Prerequisites

Required Tools
  • kubectl v1.28+
  • Helm v3.12+
  • Kubernetes cluster v1.28+
  • Container registry access
Cluster Requirements
  • Minimum 3 nodes (for HA)
  • 4 CPU, 8GB RAM per node
  • Ingress controller (nginx/traefik)
  • Storage class for PVCs

Initial Setup

namespace.yaml
apiVersion: v1
kind: Namespace
metadata:
  name: glide-hims
  labels:
    app.kubernetes.io/name: glide-hims
    app.kubernetes.io/instance: production
secrets.yaml
apiVersion: v1
kind: Secret
metadata:
  name: glide-hims-secrets
  namespace: glide-hims
type: Opaque
stringData:
  DATABASE_URL: "postgres://glide:your-password@postgres-primary:5432/glide_hims"
  REDIS_URL: "redis://:redis-password@redis-master:6379"
  JWT_SECRET: "your-jwt-secret-minimum-32-characters-long"
  LICENSE_KEY: "XXXX-XXXX-XXXX-XXXX"
---
apiVersion: v1
kind: Secret
metadata:
  name: postgres-credentials
  namespace: glide-hims
type: Opaque
stringData:
  postgres-password: "your-postgres-password"
  replication-password: "your-replication-password"
# Apply initial setup
kubectl apply -f namespace.yaml
kubectl apply -f secrets.yaml

# Or create secrets from command line
kubectl create secret generic glide-hims-secrets \
  --namespace glide-hims \
  --from-literal=JWT_SECRET=$(openssl rand -hex 32) \
  --from-literal=DATABASE_URL="postgres://..." \
  --from-literal=REDIS_URL="redis://..."

Backend Deployment

backend-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: glide-hims-backend
  namespace: glide-hims
  labels:
    app: glide-hims
    component: backend
spec:
  replicas: 3
  selector:
    matchLabels:
      app: glide-hims
      component: backend
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0
  template:
    metadata:
      labels:
        app: glide-hims
        component: backend
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "9090"
        prometheus.io/path: "/metrics"
    spec:
      serviceAccountName: glide-hims
      containers:
        - name: backend
          image: your-registry/glide-hims-backend:v1.0.0
          imagePullPolicy: Always
          ports:
            - name: http
              containerPort: 3000
            - name: metrics
              containerPort: 9090
          envFrom:
            - configMapRef:
                name: glide-hims-config
            - secretRef:
                name: glide-hims-secrets
          resources:
            requests:
              cpu: "250m"
              memory: "512Mi"
            limits:
              cpu: "1000m"
              memory: "1Gi"
          livenessProbe:
            httpGet:
              path: /api/v1/health/live
              port: http
            initialDelaySeconds: 30
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 3
          readinessProbe:
            httpGet:
              path: /api/v1/health/ready
              port: http
            initialDelaySeconds: 10
            periodSeconds: 5
            timeoutSeconds: 3
            failureThreshold: 3
          volumeMounts:
            - name: uploads
              mountPath: /app/uploads
      volumes:
        - name: uploads
          persistentVolumeClaim:
            claimName: glide-hims-uploads
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 100
              podAffinityTerm:
                labelSelector:
                  matchLabels:
                    app: glide-hims
                    component: backend
                topologyKey: kubernetes.io/hostname

Ingress

ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: glide-hims-ingress
  namespace: glide-hims
  annotations:
    kubernetes.io/ingress.class: nginx
    cert-manager.io/cluster-issuer: letsencrypt-prod
    nginx.ingress.kubernetes.io/proxy-body-size: "50m"
    nginx.ingress.kubernetes.io/proxy-read-timeout: "60"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "60"
spec:
  tls:
    - hosts:
        - hims.your-domain.com
      secretName: glide-hims-tls
  rules:
    - host: hims.your-domain.com
      http:
        paths:
          - path: /api
            pathType: Prefix
            backend:
              service:
                name: glide-hims-backend
                port:
                  number: 3000
          - path: /
            pathType: Prefix
            backend:
              service:
                name: glide-hims-frontend
                port:
                  number: 80

Horizontal Pod Autoscaler

hpa.yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: glide-hims-backend-hpa
  namespace: glide-hims
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: glide-hims-backend
  minReplicas: 3
  maxReplicas: 10
  metrics:
    - type: Resource
      resource:
        name: cpu
        target:
          type: Utilization
          averageUtilization: 70
    - type: Resource
      resource:
        name: memory
        target:
          type: Utilization
          averageUtilization: 80
  behavior:
    scaleUp:
      stabilizationWindowSeconds: 60
      policies:
        - type: Pods
          value: 2
          periodSeconds: 60
    scaleDown:
      stabilizationWindowSeconds: 300
      policies:
        - type: Pods
          value: 1
          periodSeconds: 120

PostgreSQL (Helm)

# Add Bitnami repo
helm repo add bitnami https://charts.bitnami.com/bitnami
helm repo update

# Install PostgreSQL with HA
helm install postgres bitnami/postgresql \
  --namespace glide-hims \
  --set auth.postgresPassword=your-password \
  --set auth.username=glide \
  --set auth.password=your-password \
  --set auth.database=glide_hims \
  --set primary.persistence.size=50Gi \
  --set readReplicas.replicaCount=2 \
  --set readReplicas.persistence.size=50Gi \
  --set metrics.enabled=true
postgres-values.yaml (Alternative)
# Custom values for PostgreSQL
auth:
  postgresPassword: your-secure-password
  username: glide
  password: your-secure-password
  database: glide_hims

primary:
  persistence:
    size: 50Gi
  resources:
    requests:
      cpu: 500m
      memory: 1Gi
    limits:
      cpu: 2000m
      memory: 4Gi

readReplicas:
  replicaCount: 2
  persistence:
    size: 50Gi

metrics:
  enabled: true
  serviceMonitor:
    enabled: true

backup:
  enabled: true
  cronjob:
    schedule: "0 2 * * *"
    storage:
      size: 100Gi

Redis (Helm)

# Install Redis with sentinel for HA
helm install redis bitnami/redis \
  --namespace glide-hims \
  --set auth.password=your-redis-password \
  --set architecture=replication \
  --set replica.replicaCount=2 \
  --set sentinel.enabled=true \
  --set metrics.enabled=true

Deployment Commands

# Apply all manifests
kubectl apply -f namespace.yaml
kubectl apply -f secrets.yaml
kubectl apply -f configmap.yaml
kubectl apply -f pvc.yaml
kubectl apply -f backend-deployment.yaml
kubectl apply -f frontend-deployment.yaml
kubectl apply -f services.yaml
kubectl apply -f ingress.yaml
kubectl apply -f hpa.yaml

# Or use kustomize
kubectl apply -k ./k8s/overlays/production

# Check deployment status
kubectl get pods -n glide-hims
kubectl get services -n glide-hims
kubectl get ingress -n glide-hims

# View logs
kubectl logs -f deployment/glide-hims-backend -n glide-hims

# Scale manually
kubectl scale deployment glide-hims-backend --replicas=5 -n glide-hims

# Rolling update
kubectl set image deployment/glide-hims-backend \
  backend=your-registry/glide-hims-backend:v1.1.0 \
  -n glide-hims

# Rollback
kubectl rollout undo deployment/glide-hims-backend -n glide-hims

Helm Chart (Recommended)

Install via Helm 
# Add Glide HIMS Helm repo

helm install glide-hims glide-hims/glide-hims \

  --namespace glide-hims \

  backend: 3
  frontend: 2

    repository: your-registry/glide-hims-frontend

ingress:
  enabled: true

    - host: hims.your-domain.com

Troubleshooting

# Check pod status
kubectl get pods -n glide-hims -o wide

# Describe pod for events
kubectl describe pod glide-hims-backend-xxx -n glide-hims

# View logs
kubectl logs -f pod/glide-hims-backend-xxx -n glide-hims

# Previous container logs (after crash)
kubectl logs pod/glide-hims-backend-xxx -n glide-hims --previous

# Execute into pod
kubectl exec -it pod/glide-hims-backend-xxx -n glide-hims -- sh

# Port forward for debugging
kubectl port-forward svc/glide-hims-backend 3000:3000 -n glide-hims

# Check resource usage
kubectl top pods -n glide-hims

# Check events
kubectl get events -n glide-hims --sort-by='.lastTimestamp'

# Check HPA status
kubectl get hpa -n glide-hims
