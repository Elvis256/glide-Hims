Feature Flags

Control feature availability across tenants and deployments

Enterprise Feature: Feature flags require an active enterprise license. See Licensing for details.

Overview

Feature flags allow you to:

  • Enable/disable features without code deployment
  • Roll out features gradually to specific tenants
  • A/B test new functionality
  • Control access based on license tier
  • Emergency kill switches for problematic features

Backend Usage

Using Feature Flags in Controllers 
import { Controller, Get, UseGuards } from '@nestjs/common';
import { FeatureFlagGuard, RequireFeature } from '../guards/feature-flag.guard';
import { FeatureFlagsService } from '../services/feature-flags.service';

@Controller('billing')
export class BillingController {
  constructor(private featureFlags: FeatureFlagsService) {}

  // Method 1: Decorator-based guard
  @Get('insurance-claims')
  @UseGuards(FeatureFlagGuard)
  @RequireFeature('ADVANCED_BILLING')
  async getInsuranceClaims() {
    // Only accessible if ADVANCED_BILLING is enabled
    return this.billingService.getInsuranceClaims();
  }

  // Method 2: Programmatic check
  @Get('reports')
  async getReports(@CurrentTenant() tenant: string) {
    const features = await this.featureFlags.getFeatures(tenant);
    
    if (features.isEnabled('CUSTOM_REPORTS')) {
      return this.reportService.getCustomReports();
    } else {
      return this.reportService.getStandardReports();
    }
  }
}
FeatureFlagsService 
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { FeatureFlag } from '../entities/feature-flag.entity';

@Injectable()
export class FeatureFlagsService {
  private cache = new Map<string, { features: FeatureFlag[]; timestamp: number }>();
  private CACHE_TTL = 5 * 60 * 1000; // 5 minutes

  constructor(
    @InjectRepository(FeatureFlag)
    private featureFlagRepo: Repository<FeatureFlag>,
  ) {}

  async isEnabled(tenantId: string, featureKey: string): Promise<boolean> {
    const features = await this.getFeatures(tenantId);
    const feature = features.find(f => f.key === featureKey);
    return feature?.enabled ?? false;
  }

  async getFeatures(tenantId: string): Promise<FeatureFlag[]> {
    // Check cache first
    const cached = this.cache.get(tenantId);
    if (cached && Date.now() - cached.timestamp < this.CACHE_TTL) {
      return cached.features;
    }

    // Fetch from database
    const features = await this.featureFlagRepo.find({
      where: [
        { tenantId, enabled: true },
        { tenantId: 'global', enabled: true }
      ]
    });

    // Update cache
    this.cache.set(tenantId, { features, timestamp: Date.now() });
    return features;
  }

  async setFeature(tenantId: string, key: string, enabled: boolean): Promise<void> {
    await this.featureFlagRepo.upsert(
      { tenantId, key, enabled, updatedAt: new Date() },
      ['tenantId', 'key']
    );
    
    // Invalidate cache
    this.cache.delete(tenantId);
  }
}

Frontend Usage

React Hook 
// hooks/useFeatureFlag.ts
import { useQuery } from '@tanstack/react-query';
import { api } from '../lib/api';

interface FeatureFlag {
  key: string;
  enabled: boolean;
  tier: string;
  metadata?: Record<string, any>;
}

export function useFeatureFlags() {
  return useQuery({
    queryKey: ['features'],
    queryFn: async () => {
      const response = await api.get('/features');
      return response.data.data.features as FeatureFlag[];
    },
    staleTime: 5 * 60 * 1000, // Cache for 5 minutes
  });
}

export function useFeatureFlag(featureKey: string) {
  const { data: features, isLoading } = useFeatureFlags();
  
  const feature = features?.find(f => f.key === featureKey);
  
  return {
    isEnabled: feature?.enabled ?? false,
    isLoading,
    metadata: feature?.metadata,
  };
}
Component Usage 
// components/BillingDashboard.tsx
import { useFeatureFlag } from '../hooks/useFeatureFlag';

export function BillingDashboard() {
  const { isEnabled: hasAdvancedBilling, isLoading } = useFeatureFlag('ADVANCED_BILLING');
  const { isEnabled: hasCustomReports } = useFeatureFlag('CUSTOM_REPORTS');

  if (isLoading) return <Spinner />;

  return (
    <div>
      <h1>Billing Dashboard</h1>
      
      {/* Standard billing features always shown */}
      <InvoiceList />
      <PaymentHistory />
      
      {/* Advanced billing only for enterprise */}
      {hasAdvancedBilling && (
        <>
          <InsuranceClaims />
          <BatchBilling />
          <AdvancedReports />
        </>
      )}

      {/* Feature promotion for non-enterprise */}
      {!hasAdvancedBilling && (
        <UpgradePrompt 
          feature="Advanced Billing"
          description="Get insurance claims, batch billing, and more"
        />
      )}

      {/* Custom reports section */}
      {hasCustomReports ? (
        <CustomReportBuilder />
      ) : (
        <StandardReports />
      )}
    </div>
  );
}
FeatureGate Component 
// components/FeatureGate.tsx
import { useFeatureFlag } from '../hooks/useFeatureFlag';

interface FeatureGateProps {
  feature: string;
  children: React.ReactNode;
  fallback?: React.ReactNode;
}

export function FeatureGate({ feature, children, fallback = null }: FeatureGateProps) {
  const { isEnabled, isLoading } = useFeatureFlag(feature);

  if (isLoading) return null;
  if (!isEnabled) return <>{fallback}</>;
  
  return <>{children}</>;
}

// Usage
<FeatureGate 
  feature="TELEMEDICINE" 
  fallback={<ComingSoon feature="Telemedicine" />}
>
  <TelemedicineModule />
</FeatureGate>

Database Schema

-- Feature flags table
CREATE TABLE feature_flags (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id VARCHAR(100) NOT NULL,
    key VARCHAR(100) NOT NULL,
    enabled BOOLEAN DEFAULT false,
    tier VARCHAR(50) DEFAULT 'standard',
    description TEXT,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(tenant_id, key)
);

-- System features (global defaults)
CREATE TABLE system_features (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    key VARCHAR(100) UNIQUE NOT NULL,
    name VARCHAR(200) NOT NULL,
    description TEXT,
    tier VARCHAR(50) DEFAULT 'standard',
    enabled_by_default BOOLEAN DEFAULT false,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Indexes for performance
CREATE INDEX idx_feature_flags_tenant ON feature_flags(tenant_id);
CREATE INDEX idx_feature_flags_key ON feature_flags(key);

Best Practices

Do
  • Use descriptive, uppercase flag names
  • Document each flag's purpose
  • Set appropriate cache TTLs
  • Clean up unused flags
  • Test both enabled/disabled states
  • Log flag evaluations for debugging
Don't
  • Use flags for permanent code branches
  • Create too many granular flags
  • Forget to remove deprecated flags
  • Skip feature flag in tests
  • Use flags for configuration values
  • Create circular flag dependencies
