Docker Deployment

Containerized deployment with Docker and Docker Compose

Recommended for: Development environments, small deployments, and teams familiar with Docker. For production at scale, consider Kubernetes.

Prerequisites

Required Software
  • Docker Engine 20.10+
  • Docker Compose v2.0+
  • At least 4GB RAM
  • 10GB free disk space
Install Docker
# Ubuntu/Debian
curl -fsSL https://get.docker.com | sh
sudo usermod -aG docker $USER

# Verify installation
docker --version
docker compose version

Quick Start

Get Glide HIMS running in under 5 minutes:

# Clone the repository
git clone https://github.com/your-org/glide-hims.git
cd glide-hims

# Start all services
docker compose up -d

# Check status
docker compose ps

# View logs
docker compose logs -f

Access the application at http://localhost:3000

Docker Compose Configuration

docker-compose.yml (Development) 
version: '3.8'

services:
  # PostgreSQL Database
  postgres:
    image: postgres:15-alpine
    container_name: glide-postgres
    environment:
      POSTGRES_USER: glide
      POSTGRES_PASSWORD: glide_dev_password
      POSTGRES_DB: glide_hims
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./infrastructure/docker/postgres-init.sql:/docker-entrypoint-initdb.d/init.sql
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U glide -d glide_hims"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis Cache (optional)
  redis:
    image: redis:7-alpine
    container_name: glide-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Backend API
  backend:
    build:
      context: ./packages/backend
      dockerfile: Dockerfile
      target: development
    container_name: glide-backend
    environment:
      NODE_ENV: development
      PORT: 3000
      DATABASE_URL: postgres://glide:glide_dev_password@postgres:5432/glide_hims
      REDIS_URL: redis://redis:6379
      JWT_SECRET: dev-jwt-secret-change-in-production
    volumes:
      - ./packages/backend:/app
      - /app/node_modules
    ports:
      - "3000:3000"
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    command: npm run dev

  # Frontend
  frontend:
    build:
      context: ./packages/frontend
      dockerfile: Dockerfile
      target: development
    container_name: glide-frontend
    environment:
      VITE_API_URL: http://localhost:3000/api/v1
    volumes:
      - ./packages/frontend:/app
      - /app/node_modules
    ports:
      - "5173:5173"
    depends_on:
      - backend
    command: npm run dev

volumes:
  postgres_data:
  redis_data:
docker-compose.prod.yml (Production) 
version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: glide-postgres
    restart: always
    environment:
      POSTGRES_USER: ${DB_USER:-glide}
      POSTGRES_PASSWORD: ${DB_PASSWORD:?Database password required}
      POSTGRES_DB: ${DB_NAME:-glide_hims}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER:-glide}"]
      interval: 30s
      timeout: 10s
      retries: 3

  redis:
    image: redis:7-alpine
    container_name: glide-redis
    restart: always
    command: redis-server --requirepass ${REDIS_PASSWORD:-redis_password}
    volumes:
      - redis_data:/data

  backend:
    image: glide-hims/backend:${VERSION:-latest}
    container_name: glide-backend
    restart: always
    environment:
      NODE_ENV: production
      PORT: 3000
      DATABASE_URL: postgres://${DB_USER:-glide}:${DB_PASSWORD}@postgres:5432/${DB_NAME:-glide_hims}
      REDIS_URL: redis://:${REDIS_PASSWORD:-redis_password}@redis:6379
      JWT_SECRET: ${JWT_SECRET:?JWT secret required}
      LICENSE_KEY: ${LICENSE_KEY}
      PHONE_HOME_URL: ${PHONE_HOME_URL:-https://hmisdemo.itsolutionsuganda.com}
    depends_on:
      postgres:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:3000/api/v1/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  frontend:
    image: glide-hims/frontend:${VERSION:-latest}
    container_name: glide-frontend
    restart: always
    depends_on:
      - backend

  nginx:
    image: nginx:alpine
    container_name: glide-nginx
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./infrastructure/docker/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./infrastructure/docker/ssl:/etc/nginx/ssl:ro
      - certbot_data:/var/www/certbot
    depends_on:
      - backend
      - frontend

volumes:
  postgres_data:
  redis_data:
  certbot_data:

Dockerfiles

packages/backend/Dockerfile 
# Development stage
FROM node:20-alpine AS development
WORKDIR /app
RUN npm install -g pnpm
COPY package.json pnpm-lock.yaml ./
RUN pnpm install
COPY . .
EXPOSE 3000
CMD ["pnpm", "run", "dev"]

# Build stage
FROM node:20-alpine AS builder
WORKDIR /app
RUN npm install -g pnpm
COPY package.json pnpm-lock.yaml ./
RUN pnpm install --frozen-lockfile
COPY . .
RUN pnpm run build

# Production stage
FROM node:20-alpine AS production
WORKDIR /app
RUN npm install -g pnpm

# Create non-root user
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nestjs -u 1001

COPY --from=builder /app/dist ./dist
COPY --from=builder /app/package.json ./
COPY --from=builder /app/pnpm-lock.yaml ./

RUN pnpm install --prod --frozen-lockfile

USER nestjs
EXPOSE 3000
ENV NODE_ENV=production

CMD ["node", "dist/main.js"]

Common Commands

Container Management
# Start all services
docker compose up -d

# Stop all services
docker compose down

# Restart a specific service
docker compose restart backend

# View running containers
docker compose ps

# View logs
docker compose logs -f backend

# Enter container shell
docker compose exec backend sh
Database Operations
# Access PostgreSQL CLI
docker compose exec postgres psql -U glide -d glide_hims

# Backup database
docker compose exec postgres pg_dump -U glide glide_hims > backup.sql

# Restore database
cat backup.sql | docker compose exec -T postgres psql -U glide -d glide_hims

# Run migrations
docker compose exec backend npm run migration:run

Building Production Images

# Build all images
docker compose -f docker-compose.prod.yml build

# Build specific service
docker compose -f docker-compose.prod.yml build backend

# Build with no cache (clean build)
docker compose -f docker-compose.prod.yml build --no-cache

# Tag and push to registry
docker tag glide-hims/backend:latest your-registry.com/glide-hims/backend:v1.0.0
docker push your-registry.com/glide-hims/backend:v1.0.0

# Build with version tag
VERSION=1.2.0 docker compose -f docker-compose.prod.yml build

Production Deployment Steps

1Prepare Environment 
# Copy example env and configure
cp .env.example .env
nano .env  # Edit with your production values

# Generate secure secrets
openssl rand -hex 32  # For JWT_SECRET
openssl rand -hex 16  # For DB_PASSWORD
2Set Up SSL Certificates 
# Create SSL directory
mkdir -p infrastructure/docker/ssl

# Option A: Self-signed (development/testing only)
openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout infrastructure/docker/ssl/key.pem \
  -out infrastructure/docker/ssl/cert.pem \
  -subj "/CN=localhost"

# Option B: Let's Encrypt (production)
docker run -it --rm -v $(pwd)/infrastructure/docker/ssl:/etc/letsencrypt \
  certbot/certbot certonly --standalone -d your-domain.com
3Deploy 
# Pull latest images (if using registry)
docker compose -f docker-compose.prod.yml pull

# Start services
docker compose -f docker-compose.prod.yml up -d

# Verify all services are healthy
docker compose -f docker-compose.prod.yml ps

# Check logs for any errors
docker compose -f docker-compose.prod.yml logs --tail=100
4Initialize Database 
# Run database migrations
docker compose -f docker-compose.prod.yml exec backend npm run migration:run

# Seed initial data (if needed)
docker compose -f docker-compose.prod.yml exec backend npm run seed

# Create admin user
docker compose -f docker-compose.prod.yml exec backend npm run create-admin

Health Monitoring

Built-in Health Checks
# Check all container health status
docker compose ps

# Manual health check
curl http://localhost:3000/api/v1/health

# Expected response:
{
  "status": "ok",
  "timestamp": "2026-04-10T00:00:00.000Z",
  "database": "connected",
  "redis": "connected",
  "version": "1.0.0"
}
Monitoring Script
#!/bin/bash
# health-check.sh - Run via cron every 5 minutes

HEALTH_URL="http://localhost:3000/api/v1/health"
ALERT_EMAIL="admin@example.com"

response=$(curl -s -o /dev/null -w "%{http_code}" $HEALTH_URL)

if [ "$response" != "200" ]; then
  echo "Health check failed with status $response" | \
    mail -s "Glide HIMS Alert" $ALERT_EMAIL
  
  # Optionally restart services
  docker compose -f docker-compose.prod.yml restart backend
fi

Updating Glide HIMS

# 1. Pull latest code
git pull origin main

# 2. Backup database first!
docker compose exec postgres pg_dump -U glide glide_hims > backup_$(date +%Y%m%d).sql

# 3. Rebuild images
docker compose -f docker-compose.prod.yml build

# 4. Stop and restart with new images
docker compose -f docker-compose.prod.yml up -d

# 5. Run any new migrations
docker compose -f docker-compose.prod.yml exec backend npm run migration:run

# 6. Verify health
docker compose -f docker-compose.prod.yml ps
curl http://localhost:3000/api/v1/health
Zero-Downtime Updates: For zero-downtime deployments, consider using Kubernetes with rolling updates.

Troubleshooting

# Check logs for errors
docker compose logs backend

# Common fixes:
# 1. Port already in use
sudo lsof -i :3000
# Kill the process or change port in docker-compose.yml

# 2. Permission issues
sudo chown -R $USER:$USER .

# 3. Out of disk space
docker system prune -a

# Check if postgres is running
docker compose ps postgres

# Check postgres logs
docker compose logs postgres

# Verify connectivity
docker compose exec backend nc -zv postgres 5432

# Reset database (WARNING: destroys data)
docker compose down -v
docker compose up -d

# Check resource usage
docker stats

# Increase memory limits in docker-compose.yml
services:
  backend:
    deploy:
      resources:
        limits:
          memory: 2G
        reservations:
          memory: 1G

# Enable Redis caching
REDIS_URL=redis://redis:6379
