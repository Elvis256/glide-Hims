AWS Deployment

1Launch EC2 Instance

Create an EC2 instance with Ubuntu 22.04:

# Using AWS CLI
aws ec2 run-instances \
  --image-id ami-0c7217cdde317cfec \
  --instance-type t3.medium \
  --key-name your-key-pair \
  --security-group-ids sg-xxxxxxxx \
  --subnet-id subnet-xxxxxxxx \
  --block-device-mappings '[{"DeviceName":"/dev/sda1","Ebs":{"VolumeSize":100,"VolumeType":"gp3"}}]' \
  --tag-specifications 'ResourceType=instance,Tags=[{Key=Name,Value=glide-hims}]'
Security Group Rules:
  • SSH (22) - Your IP only
  • HTTP (80) - 0.0.0.0/0
  • HTTPS (443) - 0.0.0.0/0
  • PostgreSQL (5432) - VPC CIDR only (if using external DB)
2Set Up RDS PostgreSQL (Recommended) 
# Create RDS instance
aws rds create-db-instance \
  --db-instance-identifier glide-hims-db \
  --db-instance-class db.t3.medium \
  --engine postgres \
  --engine-version 15.4 \
  --master-username glide_admin \
  --master-user-password "YourSecurePassword123!" \
  --allocated-storage 100 \
  --storage-type gp3 \
  --vpc-security-group-ids sg-xxxxxxxx \
  --db-subnet-group-name your-db-subnet-group \
  --backup-retention-period 7 \
  --multi-az \
  --storage-encrypted
Production: Enable Multi-AZ for high availability and automatic failover.
3Configure S3 for File Storage 
# Create S3 bucket for file uploads
aws s3 mb s3://glide-hims-files-your-account

# Enable versioning
aws s3api put-bucket-versioning \
  --bucket glide-hims-files-your-account \
  --versioning-configuration Status=Enabled

# Block public access
aws s3api put-public-access-block \
  --bucket glide-hims-files-your-account \
  --public-access-block-configuration \
    "BlockPublicAcls=true,IgnorePublicAcls=true,BlockPublicPolicy=true,RestrictPublicBuckets=true"
Environment Variables:
# Add to .env
AWS_REGION=us-east-1
AWS_S3_BUCKET=glide-hims-files-your-account
AWS_ACCESS_KEY_ID=AKIA...
AWS_SECRET_ACCESS_KEY=...
4Install Glide HIMS 
# SSH into EC2 instance
ssh -i your-key.pem ubuntu@ec2-xx-xx-xx-xx.compute-1.amazonaws.com

# Install dependencies
sudo apt update && sudo apt install -y nodejs npm postgresql-client nginx certbot

# Clone and install
git clone https://github.com/your-org/glide-hims.git
cd glide-hims
npm install -g pnpm pm2
pnpm install

# Configure environment
cat > packages/backend/.env << 'EOF'
NODE_ENV=production
PORT=3000
DATABASE_URL=postgres://glide_admin:YourPassword@your-rds-endpoint:5432/glide_hims
JWT_SECRET=$(openssl rand -hex 32)
AWS_REGION=us-east-1
AWS_S3_BUCKET=glide-hims-files-your-account
EOF

# Build and start
pnpm run build
pm2 start ecosystem.config.js --env production
pm2 save
5Set Up CloudFront CDN (Optional)

For better performance, set up CloudFront to cache static assets:

# Create CloudFront distribution
aws cloudfront create-distribution \
  --origin-domain-name your-domain.com \
  --default-root-object index.html

Configure cache behaviors:

  • /assets/* - Cache for 1 year (immutable)
  • /api/* - No cache, pass to origin
  • /* - Cache for 1 hour with invalidation

Google Cloud Deployment

1Create Compute Engine VM 
# Create VM instance
gcloud compute instances create glide-hims \
  --zone=us-central1-a \
  --machine-type=e2-medium \
  --image-family=ubuntu-2204-lts \
  --image-project=ubuntu-os-cloud \
  --boot-disk-size=100GB \
  --boot-disk-type=pd-ssd \
  --tags=http-server,https-server

# Create firewall rules
gcloud compute firewall-rules create allow-http \
  --allow tcp:80 --target-tags http-server

gcloud compute firewall-rules create allow-https \
  --allow tcp:443 --target-tags https-server
2Set Up Cloud SQL PostgreSQL 
# Create Cloud SQL instance
gcloud sql instances create glide-hims-db \
  --database-version=POSTGRES_15 \
  --tier=db-custom-2-4096 \
  --region=us-central1 \
  --storage-size=100GB \
  --storage-type=SSD \
  --backup-start-time=02:00 \
  --availability-type=REGIONAL

# Create database
gcloud sql databases create glide_hims --instance=glide-hims-db

# Create user
gcloud sql users create glide_admin \
  --instance=glide-hims-db \
  --password=YourSecurePassword123!
3Configure Cloud Storage 
# Create bucket
gsutil mb -l us-central1 gs://glide-hims-files

# Set lifecycle policy for old files
cat > lifecycle.json << 'EOF'
{
  "rule": [{
    "action": {"type": "SetStorageClass", "storageClass": "NEARLINE"},
    "condition": {"age": 90}
  }]
}
EOF
gsutil lifecycle set lifecycle.json gs://glide-hims-files

Azure Deployment

1Create Virtual Machine 
# Create resource group
az group create --name glide-hims-rg --location eastus

# Create VM
az vm create \
  --resource-group glide-hims-rg \
  --name glide-hims-vm \
  --image Ubuntu2204 \
  --size Standard_B2ms \
  --admin-username azureuser \
  --generate-ssh-keys \
  --public-ip-sku Standard

# Open ports
az vm open-port --port 80 --resource-group glide-hims-rg --name glide-hims-vm
az vm open-port --port 443 --resource-group glide-hims-rg --name glide-hims-vm --priority 1001
2Create Azure Database for PostgreSQL 
# Create PostgreSQL server
az postgres flexible-server create \
  --resource-group glide-hims-rg \
  --name glide-hims-db \
  --location eastus \
  --admin-user glide_admin \
  --admin-password YourSecurePassword123! \
  --sku-name Standard_B2s \
  --tier Burstable \
  --storage-size 128 \
  --version 15

# Create database
az postgres flexible-server db create \
  --resource-group glide-hims-rg \
  --server-name glide-hims-db \
  --database-name glide_hims
3Configure Azure Blob Storage 
# Create storage account
az storage account create \
  --name glidehimsfiles \
  --resource-group glide-hims-rg \
  --location eastus \
  --sku Standard_LRS \
  --kind StorageV2

# Create container
az storage container create \
  --name uploads \
  --account-name glidehimsfiles \
  --public-access off

SSL/TLS Configuration

Using Let's Encrypt (Free)
# Install certbot
sudo apt install -y certbot python3-certbot-nginx

# Obtain certificate
sudo certbot --nginx -d your-domain.com -d www.your-domain.com

# Auto-renewal is configured automatically
# Verify with:
sudo certbot renew --dry-run
Using AWS Certificate Manager
# Request certificate (for use with ALB/CloudFront)
aws acm request-certificate \
  --domain-name your-domain.com \
  --subject-alternative-names www.your-domain.com \
  --validation-method DNS

Monitoring & Logging

AWS CloudWatch
# Install CloudWatch agent
sudo apt install amazon-cloudwatch-agent

# Configure agent
cat > /opt/aws/amazon-cloudwatch-agent/etc/config.json << 'EOF'
{
  "logs": {
    "logs_collected": {
      "files": {
        "collect_list": [{
          "file_path": "/root/.pm2/logs/*.log",
          "log_group_name": "glide-hims",
          "log_stream_name": "{instance_id}"
        }]
      }
    }
  }
}
EOF

sudo systemctl start amazon-cloudwatch-agent

{

Backup Strategy

Automated Database Backups
Provider Method Retention
AWS RDS Automated snapshots 7-35 days
GCP Cloud SQL Automated backups 7-365 days
Azure Database Geo-redundant backups 7-35 days
Manual Backup Script
#!/bin/bash
# backup-to-s3.sh
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="glide_hims_$DATE.sql.gz"

# Dump and compress
pg_dump $DATABASE_URL | gzip > /tmp/$BACKUP_FILE

# Upload to S3
aws s3 cp /tmp/$BACKUP_FILE s3://glide-hims-backups/$BACKUP_FILE

# Cleanup local
rm /tmp/$BACKUP_FILE

echo "Backup completed: $BACKUP_FILE"

Cost Optimization Tips

Compute Savings
  • Reserved Instances: Up to 72% savings with 1-3 year commitment
  • Spot Instances: Up to 90% savings for dev/test environments
  • Right-sizing: Monitor and adjust instance sizes based on actual usage
  • Auto-scaling: Scale down during off-hours
Storage Savings
  • S3 Intelligent-Tiering: Automatic cost optimization
  • Lifecycle Policies: Move old data to cheaper storage
  • Database Cleanup: Archive old audit logs
  • Compression: Enable gzip for backups
